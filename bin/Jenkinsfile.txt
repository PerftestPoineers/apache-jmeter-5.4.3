pipeline {
    agent any

    tools {
        git 'Default'
    }

    stages {
        stage('Clean Workspace') {
            steps {
                cleanWs()
            }
        }

        stage('Clone Repository') {
            steps {
                git branch: 'develop', url: 'https://github.com/PerftestPoineers/apache-jmeter-5.4.3.git'
            }
        }

        stage('Run JMeter Test & Publish JMeter Report') {
            steps {
                script {
                    // Generate a unique timestamp for the filename
                    def timestamp = new Date().format('yyyyMMddHHmmss', TimeZone.getTimeZone('IST'))
                    // Define the unique results path
                    def resultsPath = "C:\\apache-jmeter-5.4.3\\bin\\results${timestamp}.jtl"
                    
                    bat """
                        "C:\\apache-jmeter-5.4.3\\bin\\jmeter.bat" -n -t "bin/Sample_Script.jmx" -l ${resultsPath}
                    """
                }
            }
        }
    }
}
